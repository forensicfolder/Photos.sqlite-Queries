SELECT

ZASSET.ZFILENAME, 
ZADDITIONALASSETATTRIBUTES.ZORIGINALFILENAME, 
ZASSET.ZDATECREATED,
datetime (ZASSET.ZDATECREATED + 978307200, 'unixepoch') AS 'Date Created',
ZASSET.ZADDEDDATE,
datetime (ZASSET.ZADDEDDATE + 978307200, 'unixepoch') AS 'Added Date',
ZASSET.ZLASTSHAREDDATE, 
datetime (ZASSET.ZLASTSHAREDDATE + 978307200, 'unixepoch') AS 'Last Shared Date',
ZASSET.ZMODIFICATIONDATE,
datetime (ZASSET.ZMODIFICATIONDATE + 978307200, 'unixepoch') AS 'Modification Date',
ZASSET.ZADJUSTMENTTIMESTAMP,
datetime (ZASSET.ZADJUSTMENTTIMESTAMP + 978307200, 'unixepoch') AS 'Adjustment Timestamp',
ZADDITIONALASSETATTRIBUTES.ZEDITORBUNDLEID,
ZADDITIONALASSETATTRIBUTES.ZASSET, 
ZASSET.ZMOMENT,
ZASSET.ZTRASHEDSTATE,
ZASSET.ZHIDDEN,
ZADDITIONALASSETATTRIBUTES.ZVIEWCOUNT, 
ZADDITIONALASSETATTRIBUTES.ZPLAYCOUNT,
ZADDITIONALASSETATTRIBUTES.ZSHARECOUNT,
ZADDITIONALASSETATTRIBUTES.ZMEDIAMETADATATYPE, 
ZASSET.ZUNIFORMTYPEIDENTIFIER,
ZASSET.ZIMPORTSESSION,
ZADDITIONALASSETATTRIBUTES.ZIMPORTEDBY, 
ZADDITIONALASSETATTRIBUTES.ZIMPORTEDBYDISPLAYNAME, 
ZADDITIONALASSETATTRIBUTES.ZIMPORTEDBYBUNDLEIDENTIFIER, 
ZADDITIONALASSETATTRIBUTES.ZORIGINALFILESIZE, 
ZASSET.ZDURATION,
ZASSET.ZLATITUDE, 
ZASSET.ZLONGITUDE, 
ZEXTENDEDATTRIBUTES.ZCAMERAMAKE, 
ZEXTENDEDATTRIBUTES.ZCAMERAMODEL,
ZASSET.ZDIRECTORY

FROM ZASSET 
LEFT JOIN ZADDITIONALASSETATTRIBUTES 
ON ZADDITIONALASSETATTRIBUTES.ZASSET = ZASSET.Z_PK 

LEFT JOIN ZEXTENDEDATTRIBUTES 
ON ZEXTENDEDATTRIBUTES.ZASSET = ZASSET.Z_PK

ORDER BY ZASSET.ZFILENAME ASC;

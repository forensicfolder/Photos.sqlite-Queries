SELECT

ZASSET.ZFILENAME, 
ZASSET.ZDIRECTORY,
ZADDITIONALASSETATTRIBUTES.ZORIGINALFILENAME, 
ZADDITIONALASSETATTRIBUTES.ZIMPORTEDBY, 
ZADDITIONALASSETATTRIBUTES.ZIMPORTEDBYDISPLAYNAME, 
ZADDITIONALASSETATTRIBUTES.ZIMPORTEDBYBUNDLEIDENTIFIER,
TS_ZCONTEXTNAMETS.ZNAME AS 'Context Name Transaction String',
ATRANSACTION.ZTIMESTAMP,
ZASSET.ZDATECREATED,
ZASSET.ZADDEDDATE,
datetime (ATRANSACTION.ZTIMESTAMP + 978307200, 'unixepoch') AS "Date and Time",
datetime (ZASSET.ZDATECREATED + 978307200, 'unixepoch') AS 'Date Created',
datetime (ZASSET.ZADDEDDATE + 978307200, 'unixepoch') AS 'Added Date'

FROM ZASSET 

LEFT JOIN ZADDITIONALASSETATTRIBUTES 
ON ZADDITIONALASSETATTRIBUTES.ZASSET = ZASSET.Z_PK 

LEFT JOIN ACHANGE
ON ACHANGE.ZENTITYPK = ZASSET.Z_PK

LEFT JOIN ATRANSACTION
ON ATRANSACTION.Z_PK = ACHANGE.ZTRANSACTIONID 

LEFT JOIN ATRANSACTIONSTRING AS TS_ZAUTHORTS
ON TS_ZAUTHORTS.Z_PK = ATRANSACTION.ZAUTHORTS

LEFT JOIN ATRANSACTIONSTRING AS TS_ZCONTEXTNAMETS
ON TS_ZCONTEXTNAMETS.Z_PK = ATRANSACTION.ZCONTEXTNAMETS

LEFT JOIN Z_PRIMARYKEY AS PK_ZENTITY
ON PK_ZENTITY.Z_ENT = ACHANGE.ZENTITY 

WHERE
ACHANGE.ZCHANGETYPE IS '0'
AND
PK_ZENTITY.Z_NAME IS 'Asset'

ORDER BY ZASSET.ZFILENAME ASC;

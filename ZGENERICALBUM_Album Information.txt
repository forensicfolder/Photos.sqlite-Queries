SELECT

ZGENERICALBUM.ZTITLE,
ZGENERICALBUM.ZCREATIONDATE,
datetime (ZGENERICALBUM.ZCREATIONDATE + 978307200, 'unixepoch') AS "Creation Date and Time",
ZGENERICALBUM.ZIMPORTEDBYBUNDLEIDENTIFIER,
ZGENERICALBUM.ZCACHEDCOUNT,
ZGENERICALBUM.ZCACHEDPHOTOSCOUNT,
ZGENERICALBUM.ZCACHEDVIDEOSCOUNT,
ZGENERICALBUM.ZTRASHEDSTATE,
ZGENERICALBUM.ZTRASHEDDATE,
datetime (ZGENERICALBUM.ZTRASHEDDATE + 978307200, 'unixepoch') AS "Trashed Date and Time",
ZGENERICALBUM.ZISOWNED,
ZGENERICALBUM.ZCLOUDMULTIPLECONTRIBUTORSENABLED,
ZGENERICALBUM.ZCLOUDMULTIPLECONTRIBUTORSENABLEDLOCAL

FROM ZGENERICALBUM

WHERE ZTITLE IS NOT NULL

ORDER BY ZCREATIONDATE ASC

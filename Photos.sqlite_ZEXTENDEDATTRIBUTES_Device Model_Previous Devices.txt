SELECT

ZASSET.ZFILENAME,
ZEXTENDEDATTRIBUTES.ZCAMERAMAKE,
ZEXTENDEDATTRIBUTES.ZCAMERAMODEL,
ZEXTENDEDATTRIBUTES.ZDATECREATED,
datetime (ZEXTENDEDATTRIBUTES.ZDATECREATED + 978307200, 'unixepoch') as 'Date Created'

FROM
ZASSET

LEFT JOIN
ZEXTENDEDATTRIBUTES
ON
ZEXTENDEDATTRIBUTES.ZASSET = ZASSET.Z_PK 

LEFT JOIN
ZADDITIONALASSETATTRIBUTES
ON
ZADDITIONALASSETATTRIBUTES.ZASSET = ZASSET.Z_PK

WHERE
ZADDITIONALASSETATTRIBUTES.ZIMPORTEDBY IN (1,2)

ORDER BY
ZEXTENDEDATTRIBUTES.ZDATECREATED ASC

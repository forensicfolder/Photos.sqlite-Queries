SELECT

ZASSET.ZFILENAME AS 'FILE NAME', 
ZGENERICALBUM.ZTITLE AS 'ALBUM NAME'
 
FROM ZASSET 

LEFT JOIN Z_28ASSETS 
ON Z_28ASSETS.Z_3ASSETS = ZASSET.Z_PK 

LEFT JOIN ZGENERICALBUM 
ON ZGENERICALBUM.Z_PK = Z_28ASSETS.Z_28ALBUMS

WHERE
ZGENERICALBUM.ZTITLE IS NOT NULL

ORDER BY ZASSET.ZFILENAME ASC

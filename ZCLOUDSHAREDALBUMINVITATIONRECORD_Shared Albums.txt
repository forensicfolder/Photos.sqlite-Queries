SELECT

ZGENERICALBUM.ZTITLE,
ZGENERICALBUM.ZCREATIONDATE,
ZCLOUDSHAREDALBUMINVITATIONRECORD.ZINVITATIONSTATE,
ZCLOUDSHAREDALBUMINVITATIONRECORD.ZISMINE,
ZCLOUDSHAREDALBUMINVITATIONRECORD.ZINVITEESUBSCRIPTIONDATE,
ZCLOUDSHAREDALBUMINVITATIONRECORD.ZALBUMGUID,
ZCLOUDSHAREDALBUMINVITATIONRECORD.ZINVITEEFIRSTNAME,
ZCLOUDSHAREDALBUMINVITATIONRECORD.ZINVITEELASTNAME,
ZCLOUDSHAREDALBUMINVITATIONRECORD.ZINVITEEFULLNAME

FROM ZCLOUDSHAREDALBUMINVITATIONRECORD 

LEFT JOIN ZGENERICALBUM
ON ZGENERICALBUM.Z_PK = ZCLOUDSHAREDALBUMINVITATIONRECORD.ZALBUM

ORDER BY ZCREATIONDATE ASC
